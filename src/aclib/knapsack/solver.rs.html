7use std::hash::Hash;

9pub fn knapsack<V, W>(n: usize, c: W, w: &[W], v: &[V]) -> V
10where
11    W: Integer + AsPrimitive<usize> + Into<usize> + Copy + Hash,
12    V: Integer + NumCast + AsPrimitive<usize> + Copy,
13{
14    if n <= 40 {
15        knapsack_half_enumerate(n, c, w, v)
16    } else if !n.overflowing_mul(c.as_()).1 {
17        let c = c.as_();
18        let w: Vec<_> = w.iter().map(|wi| wi.as_()).collect();
19        knapsack_dp_weight(n, c, &w, &v)[n][c]
20    } else {
21        let v: Vec<_> = v.iter().map(|vi| vi.as_()).collect();
22        V::from(knapsack_dp_value_solve(n, c, &w, &v)).unwrap()
23    }
24}
