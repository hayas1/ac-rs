4/// **O(2^(n/2))**, knapsack capacity is c, value of pi is v[i], weight of p_i is w[i]
5pub fn knapsack_half_enumerate<V, W>(n: usize, c: W, w: &[W], v: &[V]) -> V
6where
7    W: Integer + Copy + Hash,
8    V: Integer + Copy,
9{
10    /// **O(2^(n/2))**,compute list of tuple that (weight w, max value v such as weight is at most w)
11    fn build<V, W>(size: usize, c: W, w: &[W], v: &[V]) -> Vec<(W, V)>
12    where
13        W: Integer + Copy + Hash,
14        V: Integer + Copy,
15    {
16        let mut map = HashMap::new();
17        for i in 0..1 << size {
18            let (weight, value) = (0..size).fold((W::zero(), V::zero()), |sum, j| {
19                let (w_sum, v_sum) = sum;
20                if i >> j & 1 == 1 {
21                    (w_sum + w[j], v_sum + v[j])
22                } else {
23                    sum
24                }
25            });
26            if weight > c {
27                continue;
28            }
29            map.entry(weight).and_modify(|e: &mut V| *e = (*e).max(value)).or_insert(value);
30        }
31        let mut v: Vec<_> = map.iter().map(|(&w, &v)| (w, v)).collect();
32        v.sort_by_key(|&(wi, _vi)| wi);
33        v
34    }
35
36    let (w1, w2) = w.split_at(n / 2);
37    let (v1, v2) = v.split_at(n / 2);
38    let (map1, map2) = (build(n / 2, c, w1, v1), build(n - n / 2, c, w2, v2));
39    // TODO! this can perform not O(n^2) but O(n), however it is not bottleneck
40    // let (mut max_value, mut j) = (V::zero(), map2.len() - 1);
41    // for (w1, v1) in map1 {
42    //     // two pointers technique (in japanese shakutori-ho)
43    //     while w1 + map2[j].0 > c {
44    //         j -= if j > 0 { 1 } else { 0 };
45    //     }
46    //     max_value = max_value.max(v1 + map2[j].1);
47    // }
48    // max_value
49    map1.iter()
50        .map(|&(w1, v1)| {
51            map2.iter()
52                .map(|&(w2, v2)| if w1 + w2 <= c { v1 + v2 } else { V::zero() })
53                .max()
54                .unwrap_or(V::zero())
55        })
56        .max()
57        .unwrap_or(V::zero())
58}
