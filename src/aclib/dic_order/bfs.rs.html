4/// **O(|candidate| * (kinds of chars)^2)**, generate the kth item in dictionary order <br>
5/// warning: this use factorial, so large candidate will cause overflow <br>
6/// warning: too large k such as larger than all permutations will cause panic <br>
7pub fn kth_dic_order<T>(candidate: &[T], k: usize) -> Vec<T>
8where
9    T: Clone + Hash + Eq + Ord,
10{
11    let (factorial, mut counter) = (
12        (0..=candidate.len())
13            .scan(1, |fac, x| {
14                *fac *= x.max(1);
15                Some(*fac)
16            })
17            .collect_vec(),
18        candidate.iter().cloned().counts(),
19    );
20    let mut result = Vec::new();
21    bfs(&mut counter, k, &factorial, &mut result);
22    result
23}
24/// **O(|candidate| * |chars|^2)**, (|chars| mean HashMap's keys size), find the kth item in dictionary order
25fn bfs<T>(chars: &mut HashMap<T, usize>, k: usize, factorial: &[usize], result: &mut Vec<T>)
26where
27    T: Clone + Hash + Eq + Ord,
28{
29    if k <= 1 {
30        return result.extend(
31            chars.iter().sorted().map(|(c, &n)| vec![c.clone(); n]).flatten().collect_vec(),
32        );
33    }
34    let sorted =
35        chars.iter().filter(|&(_, &n)| n > 0).map(|(c, _)| c.clone()).sorted().collect_vec();
36    let permutations = vec![0]
37        .into_iter()
38        .chain(sorted.iter().scan(0, |cum, c| {
39            *cum += permutation(c.clone(), &chars, factorial);
40            Some(*cum)
41        }))
42        .collect_vec();
43    for (c, (&p, &np)) in sorted.iter().zip(permutations.iter().tuple_windows()) {
44        if p < k && k <= np {
45            *chars.get_mut(c).expect("above added") -= 1;
46            result.push(c.clone());
47            return bfs(chars, k - p, factorial, result);
48        }
49    }
50    unreachable!("maybe possible permutations {} < k({})", permutations[permutations.len() - 1], k);
51}
52/// **O(|chars|)** (|chars| mean HashMap's keys size), calculate permutation if given char is head
53fn permutation<T>(head: T, chars: &HashMap<T, usize>, fac: &[usize]) -> usize
54where
55    T: Clone + Hash + Eq + Ord,
56{
57    let len = chars.values().fold(0, |sum, x| sum + x);
58    if len < 2 {
59        0
60    } else {
61        let divider = chars.iter().fold(1, |prod, (c, &b)| prod * fac[b - (c == &head) as usize]);
62        fac[len - 1] / divider
63    }
64}

71    fn test_kth_dic_order1() {
72        let str: Vec<_> = "aabbcc".chars().collect();
73        assert_eq!(kth_dic_order(&str, 0), "aabbcc".chars().collect::<Vec<_>>());
74        assert_eq!(kth_dic_order(&str, 1), "aabbcc".chars().collect::<Vec<_>>());
75        assert_eq!(kth_dic_order(&str, 2), "aabcbc".chars().collect::<Vec<_>>());
