1use std::ops::{Bound, RangeBounds};
2pub trait Monoid<T> {
3    /// wrap input to Monoid
4    fn from(a: T) -> Self;
5    /// unwrap from monoid
6    fn into(self) -> T;
7    /// identity element of Monoid
8    fn identity() -> Self;
9    /// binary operation that satisfy associative law for Monoid
10    fn operation(a: &Self, b: &Self) -> Self;
11}
12
13pub struct SegmentTree<M> {
14    len: usize,
15    tree: Vec<M>, // 0-indexed perfect binary tree
16}
17impl<M> SegmentTree<M> {
18    /// **O(n)**, create segment tree. (e is identity element for a function f in type T)
19    pub fn new<T>(data: &[T]) -> Self
20    where
21        T: Clone,
22        M: Monoid<T> + Clone,
23    {
24        let binary_tree = vec![M::identity(); 2 * data.len().next_power_of_two() + 1];
25        let segment_tree = SegmentTree { len: data.len(), tree: binary_tree };
26        segment_tree.init(data)
27    }
28
29    /// **O(n)**, init segment tree by given data.
30    fn init<T>(mut self, data: &[T]) -> Self
31    where
32        T: Clone,
33        M: Monoid<T> + Clone,
34    {
35        let leaf_offset = self.leaf_offset();
36        for (i, di) in data.iter().enumerate() {
37            self.tree[leaf_offset + i] = M::from(di.clone());
38        }
39        for i in (1..leaf_offset).rev() {
40            self.tree[i] = M::operation(&self.tree[i * 2 + 1], &self.tree[i * 2 + 2]);
41        }
42        self
43    }
44
45    /// **O(1)**, return this segtree 's number of data
46    pub fn len(&self) -> usize {
47        self.len
48    }
49
50    /// **O(1)**, get beginning index of the segment tree leaf.
51    pub fn leaf_offset(&self) -> usize {
52        self.len().next_power_of_two() - 1
53    }
54
55    /// **O(log(n))**, set leaf[k] = x, and update segment tree. (non-recursive)
56    pub fn update<T>(&mut self, i: usize, x: T) -> M
57    where
58        M: Monoid<T> + Clone,
59    {
60        self.update_with(i, |_| x)
61    }
62
63    /// **O(log(n))**, update leaf[k] by f(leaf[k]), and update segment tree. (non-recursive)
64    pub fn update_with<T, U>(&mut self, i: usize, f: U) -> M
65    where
66        M: Monoid<T> + Clone,
67        U: FnOnce(&T) -> T,
68    {
69        assert!(i < self.len(), "index {} is out of 0..{}", i, self.len());
70        let mut node = self.leaf_offset() + i;
71        let mut result = M::from(f(&self.tree[node].clone().into()));
72        std::mem::swap(&mut self.tree[node], &mut result);
73        while node > 0 {
74            node = (node - 1) / 2;
75            self.tree[node] = M::operation(&self.tree[node * 2 + 1], &self.tree[node * 2 + 2]);
76        }
77        result
78    }
79
80    /// **O(1)**, range to leaf index half interval [left, right).
81    pub fn indices<R>(&self, range: R) -> (usize, usize)
82    where
83        R: RangeBounds<usize>,
84    {
85        let left = match range.start_bound() {
86            Bound::Unbounded => 0,
87            Bound::Excluded(&l) => (l + 1).max(0),
88            Bound::Included(&l) => l.max(0),
89        };
90        let right = match range.end_bound() {
91            Bound::Unbounded => self.len(),
92            Bound::Excluded(&r) => r.min(self.len()),
93            Bound::Included(&r) => (r + 1).min(self.len()),
94        };
95        assert!(left <= right);
96        (left, right)
97    }
98
99    /// **O(log(n))**, calculate f(range). (non-recursive)
100    pub fn query<T, R>(&self, range: R) -> M
101    where
102        M: Monoid<T> + Clone,
103        R: RangeBounds<usize>,
104    {
105        let (left, right) = self.indices(range);
106        let (mut left, mut right) = (self.leaf_offset() + left, self.leaf_offset() + right);
107        let (mut left_result, mut right_result) = (M::identity(), M::identity());
108        while left < right {
109            if left % 2 == 0 {
110                // l is right child
111                left_result = M::operation(&left_result, &self.tree[left]);
112                left += 1; //  move to next subtree.
113            }
114            if right % 2 == 0 {
115                // r is right child
116                right_result = M::operation(&self.tree[right - 1], &right_result);
117            }
118            left = (left - 1) / 2;
119            right = (right - 1) / 2;
120        }
121        M::operation(&left_result, &right_result)
122    }
123
124    /// **O(log^2(n))**, search the leaf where cmp(x) is true in half interval [l, r).
125    pub fn bisect<T, R, F>(&self, range: R, cmp: F, leftmost: bool) -> Option<usize>
126    where
127        R: RangeBounds<usize>,
128        M: Monoid<T> + Clone,
129        F: Fn(&T) -> bool,
130    {
131        let (mut from, mut to) = self.indices(range);
132        while to - from > 1 {
133            let mid = (from + to) / 2;
134            // bisect_right and bisect_left is merged into one function, so code is bad...
135            let (left_cmp, right_cmp) =
136                (cmp(&self.query(from..mid).into()), cmp(&self.query(mid..to).into()));
137            if leftmost && left_cmp || !leftmost && !right_cmp {
138                to = mid;
139            } else if leftmost && !left_cmp || !leftmost && right_cmp {
140                from = mid;
141            } else {
142                unreachable!();
143            }
144        }
145        if cmp(&self.tree[self.leaf_offset() + from].clone().into()) {
146            Some(from)
147        } else {
148            None
149        }
150    }
151}
