3/// **O(n log(n))**, calculate inversion number, on data such as permutation of 0,1,2,...,n-1
4pub fn inversion_number(data: &[usize]) -> usize {
5    let mut bit = vec![0; data.len()]; //fenwick tree
6    data.iter().enumerate().fold(0, |sum, (i, &di)| {
7        let prefix_sums = iter::successors(Some(di), |&j| {
8            Some(j - if j == 0 { 0 } else { 2usize.pow(j.trailing_zeros()) })
9        })
10        .take_while(|&j| 0 < j)
11        .fold(0, |p_sum, j| p_sum + bit[j]);
12        let _update_bit = iter::successors(Some(di), |&j| {
13            Some(j + 2usize.pow(if j == 0 { 0 } else { j.trailing_zeros() }))
14        })
15        .take_while(|&j| j < data.len())
16        .for_each(|j| bit[j] += 1);
17        sum + i - prefix_sums
18    })
19}

21/// **O(n log(n))**, calculate inversion number for deduplicated data
22pub fn inversion_number_with<T: Ord>(data: &[T]) -> usize {
23    let mut v: Vec<_> = (0..data.len()).collect();
24    v.sort_by(|&a, &b| data[a].cmp(&data[b]));
25    inversion_number(&v)
26}
