9pub fn from_to_indices<N, R, Nei>(neighbors: &HashMap<N, Vec<Nei>>, from_to: R) -> (N, Option<N>)
10where
11    N: Clone + Ord + Eq + Hash,
12    R: RangeBounds<N>,
13{
14    let from = match from_to.start_bound() {
15        Bound::Unbounded => panic!("from node must be specified."),
16        Bound::Excluded(l) => l.clone(),
17        Bound::Included(l) => l.clone(),
18    };
19    if !neighbors.contains_key(&from) {
20        panic!("from node must have outgoing edge.");
21    }
22    let to = match from_to.end_bound() {
23        Bound::Unbounded => None,
24        Bound::Excluded(r) => Some(r.clone()),
25        Bound::Included(r) => Some(r.clone()),
26    };
27    (from, to)
28}

30/// **O((n+m)log(n))**, shortest paths cost, with dijkstra algorithm
31pub fn dijkstra<N, R, W, C, F>(
32    neighbors: &HashMap<N, Vec<(N, W)>>,
33    from_to: R,
34    next_cost: F,
35) -> HashMap<N, C>
36where
37    N: Clone + Ord + Eq + Hash,
38    R: RangeBounds<N>,
39    C: Clone + Add + Ord + Unsigned,
40    F: Fn(C, &W) -> C,
41{
42    let (from, to) = from_to_indices(neighbors, from_to);
43    let mut costs: HashMap<_, _> = vec![(from.clone(), C::zero())].into_iter().collect();
44    let mut heap: BinaryHeap<_> = vec![Reverse((C::zero(), from))].into_iter().collect(); // for min-heap, use Reverse
45    while let Some(Reverse((cost, node))) = heap.pop() {
46        if Some(node.clone()) == to {
47            return vec![(node, cost)].into_iter().collect(); // when first come to the goal, the cost at time is minimum.
48        } else if costs.get(&node).and_then(|c| Some(cost > c.clone())).unwrap_or(false) {
49            continue; // cost is pseudo-initialized by infinite, if node has larger cost, it is skipped
50        }
51        for (nei, weight) in neighbors.get(&node).unwrap_or(&Vec::new()) {
52            let next_cost = next_cost(cost.clone(), &weight); // normal: next_cost(c, w) = c + w
53            if costs.get(nei).and_then(|c| Some(next_cost < c.clone())).unwrap_or(true) {
54                // cost is pseudo-initialized by infinite, if node has smaller cost, it is added to min-heap
55                heap.push(Reverse((next_cost.clone(), nei.clone())));
56                costs.insert(nei.clone(), next_cost);
57            }
58        }
59    }
60    costs // if loop finish, all node in connected components is visited
61}

63#[cfg(test)]
64mod test {
65    use super::*;
66
67    #[test]
68    fn test_unweighted_undirected() {
69        // 1-2
70        // \ /
71        //  0
72        // / \
73        // 3-4
74        let (_n, _m) = (5, 6);
75        let e = vec![(0, 1), (1, 2), (2, 0), (0, 3), (3, 4), (4, 0)];
76        let mut neighbors = HashMap::new();
77        for (u, v) in e {
78            neighbors.entry(u).or_insert(Vec::new()).push((v, 1u64));
79            neighbors.entry(v).or_insert(Vec::new()).push((u, 1u64));
80        }
81        assert_eq!(dijkstra(&neighbors, 1..3, |c, w| c + w), vec![(3, 2)].into_iter().collect());
82        assert_eq!(
83            dijkstra(&neighbors, 1.., |c, w| c + w),
84            vec![(0, 1), (1, 0), (2, 1), (3, 2), (4, 2)].into_iter().collect()
85        );
86        assert_eq!(dijkstra(&neighbors, 3..=4, |c: u64, w| c + w)[&4], 1);
87        // node 5 does not exist, so dijkstra search all node in the connected component
88        assert_eq!(
89            dijkstra(&neighbors, 1..5, |c, w| c + w),
90            vec![(0, 1), (1, 0), (2, 1), (3, 2), (4, 2)].into_iter().collect()
91        );
92        {
93            // silent output
94            let prev_hook = std::panic::take_hook();
95            std::panic::set_hook(Box::new(|_| {}));
96            let result = std::panic::catch_unwind(|| dijkstra(&neighbors, 5..1, |c: u64, w| c + w));
97            assert!(result.is_err()); // node 5 does not exist, so dijkstra algorithm cannot start
98            std::panic::set_hook(prev_hook);
99        }
100    }
